18  /**
19   * Scrolled text area where will be displayed the SMTP logs.
20   *
21   * @author Nilhcem
22   * @since 1.0
23   */
24  public final class LogsPane implements Observer {
25  	private final JScrollPane logsPane = new JScrollPane();
26  	private final SimpleDateFormat dateFormat = new SimpleDateFormat("hh:mm:ss a");
27  	private final JTextArea logsArea = new JTextArea();
28  
29  	/**
30  	 * Creates the text area, sets it as non-editable and sets an observer to intercept logs.
31  	 */
32  	public LogsPane() {
33  		logsArea.setEditable(false);
34  		logsPane.getViewport().add(logsArea, null);
35  		addObserverToSmtpLogAppender();
36  	}
37  
38  	/**
39  	 * Returns the JScrollPane object.
40  	 *
41  	 * @return the JScrollPane object.
42  	 */
43  	public JScrollPane get() {
44  		return logsPane;
45  	}

47  	/**
48  	 * Adds this object to the SMTP logs appender observable, to intercept logs.
49  	 * <p>
50  	 * The goal is to be informed when the log appender will received some debug SMTP logs.<br />
51  	 * When a log is written, the appender will notify this class which will display it in the text area.
52  	 * </p>
53  	 */
54  	private void addObserverToSmtpLogAppender() {
55  		Logger smtpLogger = LoggerFactory.getLogger(org.subethamail.smtp.server.Session.class);
56  		String appenderName = Configuration.INSTANCE.get("logback.appender.name");
57  
58  		@SuppressWarnings("unchecked")
59  		SMTPLogsAppender<ILoggingEvent> appender = (SMTPLogsAppender<ILoggingEvent>)
60  			((AppenderAttachable<ILoggingEvent>) smtpLogger).getAppender(appenderName);
61  		if (appender == null) {
62  			LoggerFactory.getLogger(LogsPane.class).error("Can't find logger: {}", appenderName);
63  		} else {
64  			appender.getObservable().addObserver(this);
65  		}
66  	}
67  
68  	/**
69  	 * Updates the content of the text area.
70  	 * <p>
71  	 * This method will be called by an observable element.
72  	 * <ul>
73  	 *   <li>If the observable is a {@link SMTPLogsObservable} object, the text area will display the received log.</li>
74  	 *   <li>If the observable is a {@link ClearAllButton} object, the text area will be cleared.</li>
75  	 * </ul>
76  	 * </p>
77  	 *
78  	 * @param o the observable element which will notify this class.
79  	 * @param log optional parameter (a {@code String} object, when the observable is a {@code SMTPLogsObservable} object, which will contain the log).
80  	 */
81  	@Override
82  	public void update(Observable o, Object log) {
83  		if (o instanceof SMTPLogsObservable) {
84  			// Update and scroll pane to the bottom
85  			logsArea.append(String.format("%s - %s%n", dateFormat.format(new Date()), log));
86  			logsArea.setCaretPosition(logsArea.getText().length());
87  		} else if (o instanceof ClearAllButton) {
88  			// Remove text
89  			logsArea.setText("");
90  		}
91  	}
